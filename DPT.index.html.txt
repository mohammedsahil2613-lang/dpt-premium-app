<script>
let roleSelected = '';
function showForm(role){
  roleSelected = role;
  document.getElementById('formDiv').style.display = 'block';
  if(role==='owner'){ document.getElementById('cardDiv').style.display='block'; } 
  else { document.getElementById('cardDiv').style.display='none'; }
}

function submitForm(){
  const name = document.getElementById('name').value;
  const email = document.getElementById('email').value;
  const lang = document.getElementById('language').value;

  if(!name || !email){ alert("Please fill all details"); return; }

  let amount = 0;
  if(roleSelected==='owner'){ amount=30; }
  else if(roleSelected==='investor'){ amount=20; }
  else { amount=5; }

  // PayPal link (replace YourPayPalID with your actual PayPal ID)
  const paypalLink = `https://www.paypal.com/paypalme/YourPayPalID/${amount}`;
  if(confirm(`Proceed to PayPal payment of $${amount}?`)){
    window.open(paypalLink,'_blank');
  }

  // Generate PDF
  const { jsPDF } = window.jspdf;
  const doc = new jsPDF();
  doc.text(`Role: ${roleSelected}\nName: ${name}\nEmail: ${email}\nLanguage: ${lang}`, 10, 10);
  doc.save(`${name}_DPT.pdf`);

  alert("PDF generated! Payment link opened.");
}
</script>
